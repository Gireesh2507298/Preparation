package com.example.learnvault.controller;



import com.example.learnvault.security.AuthRequest;

import com.example.learnvault.security.AuthResponse;

import com.example.learnvault.security.RegisterRequest;

import com.example.learnvault.service.AuthService;


import org.springframework.beans.factory.annotation.Autowired;

import org.springframework.http.HttpStatus;

import org.springframework.http.ResponseEntity;


import org.springframework.web.bind.annotation.*;
 
@RestController
@RequestMapping("/api/auth")
public class AuthController {

	@Autowired
	AuthService authService;
 
	@PostMapping("/register")
	public ResponseEntity<String> register(@RequestBody RegisterRequest req) {
		try {	
			String s1 = authService.signup(req);
			return new ResponseEntity<String>(s1,HttpStatus.OK);
		} catch (Exception e) {
			return ResponseEntity.badRequest().body(e.getMessage());
		}
	}
 
	@PostMapping("/login")
	public ResponseEntity<?> login(@RequestBody AuthRequest req) {
		try {
			AuthResponse ar = authService.login(req);
			return new ResponseEntity<AuthResponse>(ar,HttpStatus.OK);
		} catch (Exception e) {
			return ResponseEntity.badRequest().body(e.getMessage());
		}
	}
}

import jakarta.servlet.FilterChain;
import jakarta.servlet.ServletException;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.security.authentication
        .UsernamePasswordAuthenticationToken;
import org.springframework.security.core.authority
        .SimpleGrantedAuthority;
import org.springframework.security.core.context
        .SecurityContextHolder;
import org.springframework.security.web.authentication
        .WebAuthenticationDetailsSource;
import org.springframework.stereotype.Component;
import org.springframework.web.filter.OncePerRequestFilter;

import java.io.IOException;
import java.util.Collections;

@Component
public class JwtFilter extends OncePerRequestFilter {

    @Autowired
    private JwtUtil jwtUtil;

    @Override
    protected void doFilterInternal(
            HttpServletRequest request,
            HttpServletResponse response,
            FilterChain chain)
            throws ServletException, IOException {

        String header =
            request.getHeader("Authorization");

        String token = null;
        String email = null;
        String role  = null;

        if (header != null
                && header.startsWith("Bearer ")) {
            token = header.substring(7);
            try {
                email = jwtUtil.extractEmail(token);
                role  = jwtUtil.extractRole(token);
            } catch (Exception e) {
                System.out.println(
                    "Token error: " + e.getMessage());
            }
        }

        if (email != null
                && role != null
                && SecurityContextHolder
                    .getContext()
                    .getAuthentication() == null) {

            if (!jwtUtil.isExpired(token)) {

                // ROLE_ prefix is required
                // LEARNER → ROLE_LEARNER
                // LMS_ADMIN → ROLE_LMS_ADMIN
                SimpleGrantedAuthority authority =
                    new SimpleGrantedAuthority(
                        "ROLE_" + role);

                // print for debugging
                System.out.println(
                    "=== JWT AUTH ==="
                    + " Email:" + email
                    + " Authority:ROLE_" + role);

                UsernamePasswordAuthenticationToken
                    authToken =
                    new UsernamePasswordAuthenticationToken(
                        email,
                        null,
                        Collections.singletonList(
                            authority));

                authToken.setDetails(
                    new WebAuthenticationDetailsSource()
                        .buildDetails(request));

                SecurityContextHolder
                    .getContext()
                    .setAuthentication(authToken);
            }
        }

        chain.doFilter(request, response);
    }
}package com.example.learnvault.security;

import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.SignatureAlgorithm;
import io.jsonwebtoken.security.Keys;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Component;
import java.security.Key;
import java.util.Date;

@Component
public class JwtUtil {

    @Value("${app.jwt.secret}")
    private String secret;

    @Value("${app.jwt.expiration}")
    private long expiration;

    private Key getKey() {
        return Keys.hmacShaKeyFor(secret.getBytes());
    }

    public String generateToken(String email, String role) {
        return Jwts.builder()
                .setSubject(email)
                .claim("role", role)
                .setIssuedAt(new Date())
                .setExpiration(new Date(System.currentTimeMillis() + expiration))
                .signWith(getKey(), SignatureAlgorithm.HS256)
                .compact();
    }

    public String extractEmail(String token) {
        return getClaims(token).getSubject();
    }

    public String extractRole(String token) {
        return getClaims(token).get("role", String.class);
    }

    public boolean isExpired(String token) {
        return getClaims(token).getExpiration().before(new Date());
    }

    public boolean validateToken(String token, String email) {
        return extractEmail(token).equals(email) && !isExpired(token);
    }

    private Claims getClaims(String token) {
        return Jwts.parserBuilder()
                .setSigningKey(getKey())
                .build()
                .parseClaimsJws(token)
                .getBody();
    }
}

package com.example.learnvault.serviceimplementation;

import java.util.ArrayList;
import java.util.List;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.stereotype.Service;

import com.example.learnvault.dto.AssessmentDto;
import com.example.learnvault.entity.Assessment;
import com.example.learnvault.entity.Course;
import com.example.learnvault.entity.Module;
import com.example.learnvault.repository.AssessmentRepo;
import com.example.learnvault.repository.CourseRepo;
import com.example.learnvault.repository.ModuleRepo;
import com.example.learnvault.service.AssessmentService;

@Service
public class AssessmentServiceImpl implements AssessmentService {
	@Autowired
	AssessmentRepo assessmentRepo;
	@Autowired
	CourseRepo courseRepo;
	@Autowired
	ModuleRepo moduleRepo;

	private AssessmentDto toDto(Assessment r) {
		AssessmentDto d = new AssessmentDto();
		d.setAssessmentId(r.getAssessmentId());
		d.setCourseId(r.getCourse().getCourseId());
		if (r.getModule() != null)
			d.setModuleId(r.getModule().getModuleId());
		d.setType(r.getType());
		d.setTotalMarks(r.getTotalMarks());
		d.setPassingMarks(r.getPassingMarks());
		d.setMaxAttempts(r.getMaxAttempts());
		d.setTimeLimitMinutes(r.getTimeLimitMinutes());
		d.setStatus(r.getStatus());
		return d;
	}

	@Override
	public AssessmentDto add(AssessmentDto dto) throws Exception {
		Course c = courseRepo.findById(dto.getCourseId()).orElseThrow(() -> new Exception("Course not found"));
		Module m = dto.getModuleId() != null ? moduleRepo.findById(dto.getModuleId()).orElse(null) : null;
		Assessment r = new Assessment();
		r.setCourse(c);
		r.setModule(m);
		r.setType(dto.getType());
		r.setTotalMarks(dto.getTotalMarks());
		r.setPassingMarks(dto.getPassingMarks());
		r.setMaxAttempts(dto.getMaxAttempts());
		r.setTimeLimitMinutes(dto.getTimeLimitMinutes());
		r.setStatus(dto.getStatus());
		Assessment res = assessmentRepo.save(r);
		dto.setAssessmentId(res.getAssessmentId());
		return dto;
	}

	@Override
	public List<AssessmentDto> all() {
		List<Assessment> list = assessmentRepo.findAll();
		List<AssessmentDto> dl = new ArrayList<>();
		for (Assessment r : list)
			dl.add(toDto(r));
		return dl;
	}

	@Override
	public AssessmentDto byId(Long id) throws Exception {
		Assessment r = assessmentRepo.findById(id).orElseThrow(() -> new Exception("Assessment not found:" + id));
		return toDto(r);
	}

	@Override
	public AssessmentDto edit(AssessmentDto dto) throws Exception {
		Assessment s = assessmentRepo.findById(dto.getAssessmentId())
				.orElseThrow(() -> new Exception("Assessment not found"));
		if (dto.getStatus() != null)
			s.setStatus(dto.getStatus());
		if (dto.getTotalMarks() != null)
			s.setTotalMarks(dto.getTotalMarks());
		if (dto.getPassingMarks() != null)
			s.setPassingMarks(dto.getPassingMarks());
		assessmentRepo.save(s);
		return dto;
	}

	@Override
	public String delete(Long id) throws Exception {
		assessmentRepo.findById(id).orElseThrow(() -> new Exception("Assessment not found:" + id));
		assessmentRepo.deleteById(id);
		return "Assessment deleted";
	}

	@Override
	public List<AssessmentDto> byCourse(Long courseId) {
		List<Assessment> list = assessmentRepo.findByCourseCourseId(courseId);
		List<AssessmentDto> dl = new ArrayList<>();
		for (Assessment r : list)
			dl.add(toDto(r));
		return dl;
	}
}

@Service
public class AttemptRecordServiceImpl implements AttemptRecordService {

	@Autowired
	AttemptRecordRepo attemptRecordRepo;

	@Autowired
	AssessmentRepo assessmentRepo;

	@Autowired
	UserRepo userRepo;

	private AttemptRecordDto toDto(AttemptRecord r) {
		AttemptRecordDto d = new AttemptRecordDto();
		d.setAttemptId(r.getAttemptId());
		d.setAssessmentId(r.getAssessment().getAssessmentId());
		d.setLearnerId(r.getLearner().getUserId());
		d.setAttemptNumber(r.getAttemptNumber());
		d.setScore(r.getScore());
		d.setPassed(r.getPassed());
		d.setAttemptDate(r.getAttemptDate());
		d.setTimeTakenMinutes(r.getTimeTakenMinutes());
		return d;
	}

	@Override
	public AttemptRecordDto add(AttemptRecordDto dto) throws Exception {
		Assessment assessment = assessmentRepo.findById(dto.getAssessmentId())
				.orElseThrow(() -> new Exception("Assessment not found: " + dto.getAssessmentId()));
		User learner = userRepo.findById(dto.getLearnerId())
				.orElseThrow(() -> new Exception("Learner not found: " + dto.getLearnerId()));

		AttemptRecord r = new AttemptRecord();
		r.setAssessment(assessment);
		r.setLearner(learner);
		r.setAttemptNumber(dto.getAttemptNumber());
		r.setScore(dto.getScore());
		r.setPassed(dto.getPassed());
		r.setAttemptDate(dto.getAttemptDate());
		r.setTimeTakenMinutes(dto.getTimeTakenMinutes());

		AttemptRecord saved = attemptRecordRepo.save(r);
		dto.setAttemptId(saved.getAttemptId());
		return dto;
	}

	@Override

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.stereotype.Service;
 
import com.example.learnvault.dto.AuditLogDto;
import com.example.learnvault.entity.User;
import com.example.learnvault.enums.Role;
import com.example.learnvault.enums.UserStatus;
import com.example.learnvault.repository.UserRepo;
import com.example.learnvault.security.AuthRequest;
import com.example.learnvault.security.AuthResponse;
import com.example.learnvault.security.JwtUtil;
import com.example.learnvault.security.RegisterRequest;
import com.example.learnvault.service.AuditLogService;
import com.example.learnvault.service.AuthService;
@Service
public class AuthServiceImplementation implements AuthService {
 
	@Autowired
	UserRepo userRepo;
	@Autowired
	JwtUtil jwtUtil;
	@Autowired
	PasswordEncoder passwordEncoder;
	@Autowired
	AuditLogService auditLogService;
 
	@Override
	public AuthResponse login(AuthRequest req) {
 
	    User user = userRepo.findByEmail(req.getEmail()).orElse(null);
 
	    // ✅ Invalid email or password
	    if (user == null || !passwordEncoder.matches(req.getPassword(), user.getPassword())) {
 
	        if (user != null) {
	            try {
	                AuditLogDto log = new AuditLogDto();
	                log.setUserId(user.getUserId());
	                log.setAction("LOGIN_FAILED");
	                log.setEntityType("User");
	                auditLogService.add(log);
	            } catch (Exception ignored) {}
	        }
 
	        return new AuthResponse(null, null, null, "Invalid email or password");
	    }
 
	    // ✅ Inactive account
	    if (user.getStatus() == UserStatus.INACTIVE) {
	        return new AuthResponse(null, null, null, "Account is inactive");
	    }
 
	    // ✅ Generate token
	    String token = jwtUtil.generateToken(user.getEmail(), user.getRole().name());
 
	    // ✅ Successful login audit
	    try {
	        AuditLogDto log = new AuditLogDto();
	        log.setUserId(user.getUserId());
	        log.setAction("LOGIN_SUCCESS");
	        log.setEntityType("User");
	        auditLogService.add(log);
	    } catch (Exception ignored) {}
 
	    return new AuthResponse(token, user.getEmail(), user.getRole().name(), "Login successful");
	}
 
	@Override
	public String signup(RegisterRequest req) throws Exception {
		if (userRepo.existsByEmail(req.getEmail())) {
			throw new Exception("Email already exists");
		}
		
		User user = new User();
		user.setEmail(req.getEmail());
		user.setDepartmentId(req.getDepartmentId());
		user.setManagerId(req.getManagerId());
		user.setPassword(passwordEncoder.encode(req.getPassword()));
		user.setPhone(req.getPhone());
		user.setRole(req.getRole());
		user.setStatus(req.getUserStatus());
		user.setName(req.getName());
		User saved = userRepo.save(user);
		try {
			AuditLogDto log = new AuditLogDto();
			log.setUserId(saved.getUserId());
			log.setAction("USER_REGISTERED");
			log.setEntityType("User");
			auditLogService.add(log);
		} catch (Exception ignored) {
		}
		return "User registered successfully. Please login.";
	}
 
}

    UserRepo userRepo;

    @Mock
    PasswordEncoder passwordEncoder;

    @InjectMocks
    UserServiceImpl service;

    @Test
    void add_success() {
        UserDto dto = new UserDto();
        dto.setEmail("test@mail.com");
        dto.setPassword("1234");
        dto.setRole(Role.LEARNER);

        when(userRepo.existsByEmail("test@mail.com")).thenReturn(false);
        when(passwordEncoder.encode("1234")).thenReturn("encoded");

        User saved = new User();
        saved.setUserId(1L);

        when(userRepo.save(any(User.class))).thenReturn(saved);

        UserDto result = service.add(dto);

        assertNotNull(result.getUserId());
        verify(userRepo).save(any(User.class));
    }

    @Test
    void add_emailExists() {
        UserDto dto = new UserDto();
        dto.setEmail("exists@mail.com");

        when(userRepo.existsByEmail("exists@mail.com")).thenReturn(true);

        assertThrows(RuntimeException.class, () -> service.add(dto));
    }

    @Test
    void getAll_users() {
        User u = new User();
        u.setUserId(1L);

        when(userRepo.findAll()).thenReturn(List.of(u));

        List<UserDto> list = service.all();

        assertEquals(1, list.size());

    @Test
    void byId_success() throws Exception {
        User u = new User();
        u.setUserId(1L);

        when(userRepo.findById(1L)).thenReturn(Optional.of(u));

        UserDto dto = service.byId(1L);

        assertEquals(1L, dto.getUserId());
    }

        when(userRepo.findById(1L)).thenReturn(Optional.of(new User()));

        String res = service.delete(1L);

        verify(userRepo).deleteById(1L);
        assertEquals("User deleted", res);
    }

Project: LearnVault 
Corporate Learning & Training Management System 
Domain: Education & Learning 
1. Introduction 
LearnVault is a web-based corporate learning and training management system designed for 
enterprises, training academies, and professional certification bodies to manage course 
catalogs, learner enrollments, assessments, certifications, instructor scheduling, and learning 
analytics. The system enables employees to discover and enroll in courses, instructors to 
manage content delivery and assessments, learning administrators to track completion rates 
and certification compliance, and HR teams to align training programs with skill gap 
requirements. 
It uses a REST API-based backend architecture with Angular/React frontend and is 
compatible with Java (Spring Boot) and .NET (ASP.NET Core) platforms. 
Actors / Users 
• Learner / Employee: Browses catalog, enrolls in courses, takes assessments, earns 
certifications 
• Instructor / Trainer: Creates course content, schedules sessions, grades 
assessments 
• Learning Administrator: Manages catalog, tracks completions, and enforces 
certification compliance 
• HR Manager: Assigns mandatory training, monitors skill development, and reviews 
gap reports 
• Compliance Officer: Tracks regulatory training completions and generates 
compliance reports 
• LMS Admin: Configures learning paths, badge rules, assessment settings, and user 
roles 
2. Module Overview 
• 2.1 Identity & Access Management: Authentication, RBAC, and audit trails. 
• 2.2 Course Catalog & Content Management: Manages courses, modules, 
materials, and learning paths. 
• 2.3 Enrollment & Learning Progress: Handles course enrollments, progress 
tracking, and completions. 
• 2.4 Assessment & Evaluation: Manages quizzes, assignments, grading, and 
feedback. 
• 2.5 Certification & Badge Management: Issues certifications and digital badges on 
course completion. 
• 2.6 Instructor & Session Management: Manages instructor profiles, live sessions, 
and batch scheduling. 
• 2.7 Learning Analytics & Reporting: Provides dashboards for completion rates, 
assessment scores, and skill coverage. 
• 2.8 Notifications & Alerts: Enrollment confirmations, deadline reminders, and 
certification expiry alerts. 
3. Architecture Overview 
• Frontend: Angular or React 
• Backend: REST API-based architecture 
• Database: Relational Database (MySQL / PostgreSQL / SQL Server)

7. User Interface Design 
• Learner Portal: Course catalog, my learning dashboard, assessment centre, 
certification wallet 
• Instructor Console: Course builder, session scheduler, assessment grading, learner 
progress view 
• Learning Admin Dashboard: Enrollment tracker, completion reports, path 
assignment, badge management 
• HR Manager View: Team skill dashboard, mandatory training compliance, gap 
analysis reports 
• Compliance Portal: Regulatory training tracker, overdue report, certification validity 
dashboard 
• Admin Console: Catalog configuration, badge rule setup, user and department 
management 
8. Non-Functional Requirements 
• Performance: Support 25,000+ concurrent learners during mandatory training 
windows 
• Security: Content access control, RBAC, secure assessment delivery to prevent 
answer leakage 
• Scalability: Supports multi-tenant deployment for multiple business units or client 
organizations 
• Availability: 99.9% uptime with offline progress sync capability for unreliable network 
environments 
• Maintainability: Modular course structure, API versioning, configurable assessment 
and badge rules 
• Observability: Real-time learning dashboards, completion pipeline metrics, and 
compliance alerts 
9. Assumptions & Constraints 
• Phase 1 excludes video streaming hosting; content URLs reference internally hosted 
static files 
• Live session delivery uses internal virtual room identifiers; video conferencing 
integration is deferred 
• All notifications are in-app only; email and calendar integrations are out of Phase 1 
scope 
• Assessment anti-cheat mechanisms (proctoring) are out of scope for Phase 1 
• Fully implementable using standard Java Spring Boot / ASP.NET Core, 
Angular/React, and RDBMS tooling
